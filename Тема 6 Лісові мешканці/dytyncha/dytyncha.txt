<script>
const PAIRS = [
  { id: "p1", adultSrc: "im1.png", babySrc: "im1.1.png" }, 
  { id: "p2", adultSrc: "im2.png", babySrc: "im2.1.png" }, 
  { id: "p3", adultSrc: "im3.png", babySrc: "im3.1.png" }, 
  
  { id: "p4", adultSrc: "im4.png", babySrc: "im4.1.png" }, 
  { id: "p5", adultSrc: "im5.png", babySrc: "im5.1.png" }, 
  { id: "p6", adultSrc: "im6.png", babySrc: "im6.1.png" }, 

  { id: "p7", adultSrc: "im7.png", babySrc: "im7.1.png" }, 
];

const els = {
  colLeft: document.getElementById('colLeft'),
  colCenter: document.getElementById('colCenter'),
  colRight: document.getElementById('colRight'),
  pool: document.getElementById('babiesPool'),
  winMsg: document.getElementById('winMsg'),
  resetBtn: document.getElementById('resetBtn')
};

let selectedBaby = null;

function initGame() {
  els.colLeft.innerHTML = '';
  els.colRight.innerHTML = '';
  els.pool.innerHTML = '';
  
  const centerRows = els.colCenter.querySelectorAll('.row');
  centerRows.forEach(r => r.remove());

  els.winMsg.classList.remove('show');
  selectedBaby = null;

  const leftPairs = PAIRS.slice(0, 3);
  const rightPairs = PAIRS.slice(3, 6);
  const owlPair = PAIRS[6];

  leftPairs.forEach(p => createAdultRow(p, els.colLeft, false));
  rightPairs.forEach(p => createAdultRow(p, els.colRight, true)); 
  createAdultRow(owlPair, els.colCenter, false, true); 

  const shuffled = [...PAIRS].sort(() => Math.random() - 0.5);
  shuffled.forEach(p => createBaby(p));
}

function createAdultRow(pair, container, isReverse, isCenter = false) {
  const row = document.createElement('div');
  row.className = `row ${isReverse ? 'reverse' : ''} ${isCenter ? 'center-style' : ''}`;
  row.dataset.id = pair.id;

  const img = document.createElement('img');
  img.src = pair.adultSrc;
  img.className = 'adult-img';
  
  const slot = document.createElement('div');
  slot.className = 'slot';
  
  slot.addEventListener('click', () => handleSlotClick(slot, row));
  
  slot.addEventListener('dragover', e => {
    e.preventDefault();
    slot.classList.add('highlight');
  });
  slot.addEventListener('dragleave', () => slot.classList.remove('highlight'));
  slot.addEventListener('drop', e => {
    e.preventDefault();
    slot.classList.remove('highlight');
    const babyId = e.dataTransfer.getData('text');
    checkMatch(babyId, row, slot);
  });

  row.appendChild(img);
  row.appendChild(slot);
  container.appendChild(row);
}

function createBaby(pair) {
  const img = document.createElement('img');
  img.src = pair.babySrc;
  img.className = 'baby';
  img.dataset.id = pair.id;
  img.draggable = true;

  img.addEventListener('click', (e) => {
    e.stopPropagation();
    selectBaby(img);
  });

  img.addEventListener('dragstart', e => {
    selectBaby(img);
    e.dataTransfer.setData('text', pair.id);
  });

  els.pool.appendChild(img);
}

function selectBaby(imgNode) {
  if (imgNode.parentElement.classList.contains('slot')) return;
  document.querySelectorAll('.baby.selected').forEach(b => b.classList.remove('selected'));
  imgNode.classList.add('selected');
  selectedBaby = imgNode;
  
  document.querySelectorAll('.row:not(.matched) .slot').forEach(s => {
    s.classList.add('highlight');
    setTimeout(() => s.classList.remove('highlight'), 300);
  });
}

function handleSlotClick(slot, row) {
  if (!selectedBaby) return; 
  if (row.classList.contains('matched')) return; 
  checkMatch(selectedBaby.dataset.id, row, slot);
}

function checkMatch(babyId, row, slot) {
  const babyEl = document.querySelector(`.baby[data-id="${babyId}"]`);
  if (!babyEl) return;

  if (babyId === row.dataset.id) {
    slot.appendChild(babyEl);
    row.classList.add('matched');
    babyEl.classList.remove('selected');
    babyEl.draggable = false;
    selectedBaby = null;
    checkWinCondition();
  } else {
    babyEl.classList.add('shake');
    if (navigator.vibrate) navigator.vibrate(200);
    setTimeout(() => babyEl.classList.remove('shake'), 400);
  }
}

function checkWinCondition() {
  const matchedCount = document.querySelectorAll('.row.matched').length;
  if (matchedCount === PAIRS.length) {
    els.winMsg.classList.add('show');
    els.winMsg.scrollIntoView({ behavior: 'smooth', block: 'center' });
  }
}

document.addEventListener('click', (e) => {
  if (!e.target.closest('.baby') && !e.target.closest('.slot')) {
    if (selectedBaby) {
        selectedBaby.classList.remove('selected');
        selectedBaby = null;
    }
  }
});

els.resetBtn.addEventListener('click', initGame);

initGame();
</script>
