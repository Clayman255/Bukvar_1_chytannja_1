        <button id="check-btn">Перевірити</button>
        <button id="reset-btn">Ще раз</button>
    </div>

    <div id="message"></div>

    <script>
        const container = document.getElementById('game-container');
        const checkBtn = document.getElementById('check-btn');
        const resetBtn = document.getElementById('reset-btn');
        const message = document.getElementById('message');

        const correctOrder = ['1.png', '2.png', '3.png', '4.png', '5.png'];

        function highlightCorrect() {
            const cards = [...container.querySelectorAll('.img-card')];
            cards.forEach((card, index) => {
                if (card.dataset.filename === correctOrder[index]) {
                    card.classList.add('at-correct-place');
                } else {
                    card.classList.remove('at-correct-place');
                }
            });
        }

        function initGame() {
            message.textContent = '';
            container.innerHTML = '';
            
            // Створюємо перемішаний масив без першої картинки
            let restOrder = ['2.png', '3.png', '4.png', '5.png'].sort(() => Math.random() - 0.5);
            
            // Перша картинка завжди на початку
            let displayOrder = ['1.png', ...restOrder];

            displayOrder.forEach((src) => {
                const card = document.createElement('div');
                card.className = 'img-card';
                card.draggable = true;
                card.dataset.filename = src;
                
                const img = document.createElement('img');
                img.src = src;
                img.alt = 'Сюжетна картинка';
                
                card.appendChild(img);
                
                card.addEventListener('dragstart', handleDragStart);
                card.addEventListener('dragover', handleDragOver);
                card.addEventListener('drop', handleDrop);
                card.addEventListener('dragend', handleDragEnd);
                
                container.appendChild(card);
            });
            highlightCorrect();
        }

        let draggedItem = null;

        function handleDragStart(e) {
            draggedItem = this;
            this.classList.add('dragging');
            e.dataTransfer.effectAllowed = 'move';
        }

        function handleDragOver(e) {
            e.preventDefault();
            e.dataTransfer.dropEffect = 'move';
        }

        function handleDrop(e) {
            e.preventDefault();
            if (this !== draggedItem) {
                const allCards = [...container.querySelectorAll('.img-card')];
                const draggedIdx = allCards.indexOf(draggedItem);
                const targetIdx = allCards.indexOf(this);

                if (draggedIdx < targetIdx) {
                    this.after(draggedItem);
                } else {
                    this.before(draggedItem);
                }
                highlightCorrect();
            }
        }

        function handleDragEnd() {
            this.classList.remove('dragging');
            draggedItem = null;
        }

        checkBtn.addEventListener('click', () => {
            const userOrder = [...container.querySelectorAll('.img-card')].map(card => card.dataset.filename);
            const isCorrect = userOrder.every((val, index) => val === correctOrder[index]);

            if (isCorrect) {
                message.textContent = 'Молодець! Правильно!';
                message.className = 'correct';
            } else {
                message.textContent = 'Спробуй ще раз!';
                message.className = 'wrong';
            }
        });

        resetBtn.addEventListener('click', initGame);

        window.onload = initGame;
    </script>
