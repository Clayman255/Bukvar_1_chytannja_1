<script>
  const animals = [
    {id:1, img:"im1.png", name:"Вовк", correct:true,  big:true},
    {id:2, img:"im2.png", name:"Ведмідь", correct:true, big:true},
    {id:3, img:"im3.png", name:"Сова", correct:true,    big:false},
    {id:4, img:"im4.png", name:"Соловей", correct:true, big:false},
    {id:5, img:"im5.png", name:"Олень", correct:false,  big:true},
    {id:6, img:"im6.png", name:"Лисиця", correct:false, big:true}
  ];

  const scene = document.getElementById("scene");
  const foundList = document.getElementById("foundList");
  const winMsg = document.getElementById("winMsg");
  const audioTask = document.getElementById("audioTask");

  let correctCount = 0;
  const totalCorrect = animals.filter(a => a.correct).length;

  function shuffle(arr){ return [...arr].sort(() => Math.random() - 0.5); }

  function initGame(){
    scene.innerHTML = "";
    foundList.innerHTML = "";
    winMsg.style.display = "none";
    correctCount = 0;

    const shuffled = shuffle(animals);

    shuffled.forEach(a => {
      const cell = document.createElement("div");
      cell.className = "cell";
      
      const el = document.createElement("div");
      el.className = "animal" + (a.big ? " big" : "");
      el.innerHTML = `<img src="${a.img}"><div class="label">${a.name}</div>`;
      
      el.onclick = () => {
        if(a.correct){
          el.style.visibility = "hidden";
          document.getElementById("soundCorrect").play();
          addFound(a);
          if(++correctCount === totalCorrect) winMsg.style.display = "block";
        } else {
          el.classList.add("error");
          document.getElementById("soundWrong").play();
          setTimeout(() => el.classList.remove("error"), 600);
        }
      };

      cell.appendChild(el);
      scene.appendChild(cell);
    });
  }

  function addFound(a){
    const div = document.createElement("div");
    div.style.display = "flex";
    div.style.alignItems = "center";
    div.style.background = "#fff";
    div.style.padding = "5px";
    div.style.borderRadius = "10px";
    div.style.marginBottom = "5px";

    /* ЗБІЛЬШЕНО ВДВІЧІ: зображення (40 -> 80) і напис (додано font-size ~2x) */
    div.innerHTML = `<img src="${a.img}" width="80"> <span style="margin-left:10px; font-weight:bold; font-size:2rem;">${a.name}</span>`;
    foundList.appendChild(div);
  }

  document.getElementById("btnSound").onclick = () => audioTask.play();
  document.getElementById("btnReset").onclick = initGame;

  initGame();
</script>
