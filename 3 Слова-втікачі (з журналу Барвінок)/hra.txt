<script>
  const data = [
    { id: 1, textBefore: "У саду росте", textAfter: ".", answer: "яблуня" },
    { id: 2, textBefore: "Мій", textAfter: "називається Мурчик.", answer: "кіт" },
    { id: 3, textBefore: "У мого друга є", textAfter: "песик.", answer: "маленький" },
    { id: 4, textBefore: "На мій день народження", textAfter: "спекла торт.", answer: "мама" },
    { id: 5, textBefore: "Мені подобається", textAfter: "в теніс.", answer: "грати" }
  ];

  const $pool = document.getElementById("pool");
  const $list = document.getElementById("list");
  const $win = document.getElementById("win");
  const sndOk = document.getElementById("sound-ok");
  const sndNo = document.getElementById("sound-no");

  function init() {
    // Render words on the left
    const shuffledAnswers = data.map(d => d.answer).sort(() => Math.random() - 0.5);
    shuffledAnswers.forEach(word => {
      const el = document.createElement("div");
      el.className = "tile";
      el.textContent = word;
      el.draggable = true;
      el.addEventListener("dragstart", (e) => {
        e.dataTransfer.setData("text", word);
        e.target.classList.add("dragging");
      });
      $pool.appendChild(el);
    });

    // Render sentences on the right
    data.forEach(item => {
      const card = document.createElement("div");
      card.className = "sentence-card";
      
      const before = document.createElement("span");
      before.textContent = item.textBefore;
      
      const zone = document.createElement("div");
      zone.className = "drop-zone";
      zone.dataset.answer = item.answer;
      
      const after = document.createElement("span");
      after.textContent = item.textAfter;

      card.appendChild(before);
      card.appendChild(zone);
      card.appendChild(after);
      $list.appendChild(card);

      // Drag and Drop events
      zone.addEventListener("dragover", (e) => e.preventDefault());
      zone.addEventListener("dragenter", () => zone.classList.add("over"));
      zone.addEventListener("dragleave", () => zone.classList.remove("over"));
      zone.addEventListener("drop", (e) => {
        e.preventDefault();
        zone.classList.remove("over");
        const droppedWord = e.dataTransfer.getData("text");
        
        if (droppedWord === zone.dataset.answer) {
          zone.textContent = droppedWord;
          zone.classList.add("locked");
          sndOk.play();
          markWordUsed(droppedWord);
          checkWin();
        } else {
          sndNo.play();
          zone.classList.add("bad-shake");
          setTimeout(() => zone.classList.remove("bad-shake"), 500);
        }
      });
    });
  }

  function markWordUsed(word) {
    const tiles = document.querySelectorAll(".tile");
    for (let t of tiles) {
      if (t.textContent === word && !t.classList.contains("used")) {
        t.classList.add("used");
        t.draggable = false;
        break;
      }
    }
  }

  function checkWin() {
    const total = data.length;
    const solved = document.querySelectorAll(".drop-zone.locked").length;
    if (solved === total) {
      setTimeout(() => $win.style.display = "flex", 600);
    }
  }

  init();
</script>
