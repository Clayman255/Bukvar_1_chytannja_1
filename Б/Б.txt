<script>
const levels = [
  { word:"робітник", syllables:["ро","біт","ник"] },
  { word:"барбарис", syllables:["бар","ба","рис"] },
  { word:"барабан", syllables:["ба","ра","бан"] },
  { word:"субота", syllables:["су","бо","та"] },
  { word:"борсук", syllables:["бор","сук"] },
  { word:"будинок", syllables:["бу","ди","нок"] },
];

const game = document.getElementById("game");
const afterWin = document.getElementById("afterWin");
const au = document.getElementById("au");
const ok = document.getElementById("ok");
const no = document.getElementById("no");

let dragged = null;

function shuffle(arr){
  return [...arr].sort(()=>Math.random()-0.5);
}

function init(){
  game.innerHTML="";
  afterWin.style.display="none";

  levels.forEach((lvl, index)=>{
    const row=document.createElement("div");
    row.className="task-row";
    row.dataset.idx = index; // Зберігаємо індекс, щоб знайти слово пізніше

    const left=document.createElement("div");
    left.className="drop-zone";

    const slots=document.createElement("div");
    slots.className="slots";

    lvl.syllables.forEach((s)=>{
      const slot=document.createElement("div");
      slot.className="slot";
      slot.dataset.expected=s;
      slot.addEventListener("dragover",e=>e.preventDefault());
      slot.addEventListener("drop",()=>drop(slot,row));
      slots.appendChild(slot);
    });

    left.appendChild(slots);

    const right=document.createElement("div");
    right.className="syllables";

    // Мішаємо так, щоб перший не був першим (якщо можливо)
    let parts = [...lvl.syllables];
    if(parts.length > 1) {
       const first = parts[0];
       const rest = shuffle(parts.slice(1));
       parts = [...rest, first];
    } else {
       parts = shuffle(parts);
    }

    parts.forEach(s=>{
      const el=document.createElement("div");
      el.className="syllable";
      el.textContent=s;
      el.draggable=true;
      el.addEventListener("dragstart",()=>dragged=el);
      right.appendChild(el);
    });

    row.appendChild(left);
    row.appendChild(right);
    game.appendChild(row);
  });
}

function drop(slot,row){
  if(!dragged || slot.children.length) return;

  if(dragged.textContent!==slot.dataset.expected){
    no.play();
    return;
  }

  slot.appendChild(dragged);
  dragged.classList.add("used");
  dragged.draggable=false;
  dragged=null;
  ok.play();

  checkRow(row);
}

function checkRow(row){
  const slots=row.querySelectorAll(".slot");
  // Якщо всі слоти заповнені
  if([...slots].every(s=>s.children.length)){
    
    // Отримуємо повне слово з масиву
    const idx = row.dataset.idx;
    const fullWord = levels[idx].word;

    // ПЕРЕТВОРЕННЯ:
    // 1. Очищаємо вміст рядка
    row.innerHTML = "";
    
    // 2. Додаємо клас для центрування
    row.classList.add("completed-text");
    row.classList.add("correct-row"); // мітка, що рядок готовий

    // 3. Вставляємо просто слово
    const textDiv = document.createElement("div");
    textDiv.className = "final-text";
    textDiv.textContent = fullWord;
    row.appendChild(textDiv);

    checkWin();
  }
}

function checkWin(){
  // Перевіряємо, чи кількість готових рядків дорівнює загальній кількості рівнів
  if(document.querySelectorAll(".correct-row").length === levels.length){
    afterWin.style.display="flex";
    
    // Піднімаємо сторінку вгору плавно
    window.scrollTo({
      top: 0,
      behavior: 'smooth'
    });
  }
}

document.getElementById("playSound").onclick=()=>au.play();
document.getElementById("again").onclick=init;

init();
</script>
