<script>
  let currentColor = null;

  function selectColor(color, btn) {
    currentColor = color;
    // Знімаємо виділення з усіх кнопок палітри
    document.querySelectorAll('.color-btn').forEach(b => b.classList.remove('active'));
    // Виділяємо натиснуту
    btn.classList.add('active');
  }

  function paintCircle(circle) {
    if (!currentColor) {
      alert("Спочатку обери колір на палітрі внизу!");
      return;
    }

    const card = circle.closest('.card');
    const correctColor = card.dataset.correct;

    if (currentColor === correctColor) {
      // Фарбуємо в обраний колір
      circle.style.backgroundColor = (currentColor === 'green') ? '#22c55e' : '#ef4444';
      circle.style.borderColor = 'white';
      circle.classList.add('painted');
      checkWin();
    } else {
      // Анімація помилки (трясіння)
      card.style.borderColor = '#ef4444';
      card.animate([
        { transform: 'translateX(-5px)' },
        { transform: 'translateX(5px)' }
      ], { duration: 100, iterations: 4 });
      setTimeout(() => card.style.borderColor = 'transparent', 500);
    }
  }

  function checkWin() {
    const painted = document.querySelectorAll('.circle-slot.painted');
    if (painted.length === 4) {
      setTimeout(() => {
        document.getElementById('winMsg').style.display = 'block';
      }, 500);
    }
  }
</script>
